function cekProxy() { const proxyInput = document.getElementById('proxyInput').value; const hasil = document.getElementById('hasil'); const [proxyAddress, proxyPort] = proxyInput.split(':'); const url = 'http://www.google.com'; // URL yang akan diakses const proxyUrl = `http://${proxyAddress}:${proxyPort}`; // Membuat objek Request untuk proxy const proxyRequest = new Request(proxyUrl, { method: 'GET', mode: 'no-cors', cache: 'no-cache', credentials: 'same-origin', headers: { 'User-Agent': 'Mozilla/5.0' }, redirect: 'follow', referrerPolicy: 'no-referrer' }); // Melakukan permintaan melalui proxy menggunakan fetch fetch(proxyRequest) .then((response) => { if (response.ok) { hasil.textContent = 'Proxy masih hidup.'; } else { hasil.textContent = 'Proxy mati atau tidak merespons dengan benar.'; } }) .catch((error) => { hasil.textContent = 'Terjadi kesalahan saat mencoba terhubung ke proxy: ' + error.message; }); }
