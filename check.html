const http = require('http'); const proxyAddress = '212.83.143.60'; const proxyPort = 58239; const options = { host: proxyAddress, port: proxyPort, path: 'http://www.google.com', // Coba mengakses Google sebagai tes timeout: 5000 // Batas waktu dalam milidetik }; const req = http.request(options, (res) => { if (res.statusCode === 200) { console.log('Proxy masih hidup.'); } else { console.log('Proxy mati atau tidak merespons dengan benar.'); } }); req.on('error', (err) => { console.error('Terjadi kesalahan saat mencoba terhubung ke proxy:', err.message); }); req.end();
